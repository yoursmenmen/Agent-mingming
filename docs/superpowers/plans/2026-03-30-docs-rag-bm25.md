package com.mingming.agent.rag;

import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.security.MessageDigest;
import java.util.ArrayList;
import java.util.HexFormat;
import java.util.List;
import org.springframework.stereotype.Service;

@Service
public class DocsChunkingService {

    public List<DocsChunk> loadChunks(Path docsRoot) {
        try {
            List<DocsChunk> all = new ArrayList<>();
            Files.walk(docsRoot)
                    .filter(path -> path.toString().endsWith(".md"))
                    .forEach(path -> {
                        try {
                            all.addAll(chunkMarkdown(docsRoot.relativize(path), Files.readString(path)));
                        } catch (Exception ignored) {
                        }
                    });
            return all;
        } catch (Exception ex) {
            return List.of();
        }
    }

    public List<DocsChunk> chunkMarkdown(Path relativePath, String markdown) {
        List<DocsChunk> chunks = new ArrayList<>();
        String currentHeading = "未命名章节";
        StringBuilder buffer = new StringBuilder();
        int ordinal = 0;

        for (String line : markdown.split("\\R")) {
            if (line.startsWith("#")) {
                if (!buffer.toString().isBlank()) {
                    chunks.add(buildChunk(relativePath, currentHeading, buffer.toString(), ordinal++));
                    buffer.setLength(0);
                }
                currentHeading = line.replaceFirst("^#+\\s*", "").trim();
                continue;
            }
            if (line.isBlank() && buffer.length() > 320) {
                chunks.add(buildChunk(relativePath, currentHeading, buffer.toString(), ordinal++));
                buffer.setLength(0);
                continue;
            }
            buffer.append(line).append('\n');
        }

        if (!buffer.toString().isBlank()) {
            chunks.add(buildChunk(relativePath, currentHeading, buffer.toString(), ordinal));
        }
        return chunks;
    }

    private DocsChunk buildChunk(Path relativePath, String heading, String content, int ordinal) {
        String docPath = relativePath.toString().replace('\\', '/');
        String normalized = content.trim();
        String id = stableHash(docPath + "|" + heading + "|" + ordinal + "|" + normalized);
        int tokenEstimate = Math.max(1, normalized.length() / 2);
        return new DocsChunk(id, docPath, heading, normalized, tokenEstimate);
    }

    private String stableHash(String input) {
        try {
            MessageDigest digest = MessageDigest.getInstance("SHA-256");
            return HexFormat.of().formatHex(digest.digest(input.getBytes(StandardCharsets.UTF_8))).substring(0, 16);
        } catch (Exception ex) {
            return Integer.toHexString(input.hashCode());
        }
    }
}

package com.mingming.agent.rag;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import org.springframework.stereotype.Service;

@Service
public class Bm25RetrieverService {

    public record RetrievalHit(DocsChunk chunk, double score) {}

    public List<RetrievalHit> search(String query, List<DocsChunk> chunks, int topK, double threshold) {
        List<String> queryTerms = tokenize(query);
        if (queryTerms.isEmpty() || chunks.isEmpty()) {
            return List.of();
        }

        Map<String, Integer> docFreq = new HashMap<>();
        List<List<String>> tokenized = new ArrayList<>();
        double avgLen = 0.0d;

        for (DocsChunk chunk : chunks) {
            List<String> terms = tokenize(chunk.content());
            tokenized.add(terms);
            avgLen += terms.size();
            terms.stream().distinct().forEach(term -> docFreq.merge(term, 1, Integer::sum));
        }
        avgLen = Math.max(1.0d, avgLen / chunks.size());

        List<RetrievalHit> hits = new ArrayList<>();
        for (int i = 0; i < chunks.size(); i++) {
            double score = bm25Score(queryTerms, tokenized.get(i), docFreq, chunks.size(), avgLen);
            if (score >= threshold) {
                hits.add(new RetrievalHit(chunks.get(i), score));
            }
        }

        hits.sort((a, b) -> Double.compare(b.score(), a.score()));
        return hits.stream().limit(topK).toList();
    }

    private double bm25Score(List<String> queryTerms, List<String> docTerms, Map<String, Integer> df, int n, double avgLen) {
        double k1 = 1.2d;
        double b = 0.75d;
        Map<String, Integer> tf = new HashMap<>();
        for (String term : docTerms) {
            tf.merge(term, 1, Integer::sum);
        }

        double score = 0.0d;
        for (String q : queryTerms) {
            int freq = tf.getOrDefault(q, 0);
            if (freq == 0) {
                continue;
            }
            int docFreq = df.getOrDefault(q, 0);
            double idf = Math.log(1 + (n - docFreq + 0.5d) / (docFreq + 0.5d));
            double denom = freq + k1 * (1 - b + b * docTerms.size() / avgLen);
            score += idf * (freq * (k1 + 1) / denom);
        }
        return score;
    }

    private List<String> tokenize(String text) {
        String normalized = text == null ? "" : text.toLowerCase();
        List<String> terms = new ArrayList<>();
        for (String token : normalized.split("[^\\p{L}\\p{N}]+")) {
            if (!token.isBlank()) {
                terms.add(token);
                if (token.length() >= 2) {
                    for (int i = 0; i < token.length() - 1; i++) {
                        terms.add(token.substring(i, i + 2));
                    }
                }
            }
        }
        return terms;
    }
}

package com.mingming.agent.rag;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.ArrayNode;
import com.fasterxml.jackson.databind.node.ObjectNode;
import com.mingming.agent.entity.RunEventEntity;
import com.mingming.agent.event.RunEventType;
import com.mingming.agent.repository.RunEventRepository;
import java.time.OffsetDateTime;
import java.util.List;
import java.util.UUID;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

@Service
@RequiredArgsConstructor
public class RetrievalEventService {
    private final RunEventRepository runEventRepository;
    private final ObjectMapper objectMapper;

    public void record(UUID runId, int seq, String query, int topK, int topN, List<Bm25RetrieverService.RetrievalHit> hits) {
        ObjectNode payload = objectMapper.createObjectNode();
        payload.put("query", query);
        payload.put("topK", topK);
        payload.put("topN", topN);
        ArrayNode items = payload.putArray("hits");
        for (Bm25RetrieverService.RetrievalHit hit : hits) {
            ObjectNode n = items.addObject();
            n.put("chunkId", hit.chunk().chunkId());
            n.put("docPath", hit.chunk().docPath());
            n.put("headingPath", hit.chunk().headingPath());
            n.put("score", hit.score());
            n.put("snippet", hit.chunk().content().substring(0, Math.min(120, hit.chunk().content().length())));
        }

        RunEventEntity entity = new RunEventEntity();
        entity.setId(UUID.randomUUID());
        entity.setRunId(runId);
        entity.setSeq(seq);
        entity.setCreatedAt(OffsetDateTime.now());
        entity.setType(RunEventType.RETRIEVAL_RESULT.name());
        entity.setPayload(payload.toString());
        runEventRepository.save(entity);
    }
}
